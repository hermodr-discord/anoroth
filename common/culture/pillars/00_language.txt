language_anorothic = {
	type = language
	is_shown = {
		language_is_shown_trigger = {
			LANGUAGE = language_anorothic
		}
	}
	ai_will_do = {
		value = 10
		if = {
			limit = { has_cultural_pillar = language_anorothic }
			multiply = 10
		}
	}
	
	color = anorothic
}
